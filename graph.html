Graph Algorithms and Data Structures

Graph Implementation using Adjacency List

class Graph:
    def __init__(self, vertices):
        self.vertices = vertices
        self.graph = {i: [] for i in range(vertices)}

    def add_edge(self, u, v):
        self.graph[u].append(v)
        self.graph[v].append(u)

Graph Implementation using Adjacency Matrix

class Graph:
    def __init__(self, vertices):
        self.vertices = vertices
        self.graph = [[0] * vertices for _ in range(vertices)]

    def add_edge(self, u, v):
        self.graph[u][v] = 1
        self.graph[v][u] = 1

Breadth-First Search (BFS) in a Graph

from collections import deque

def bfs(graph, start):
    visited = [False] * len(graph)
    queue = deque([start])
    visited[start] = True
    while queue:
        vertex = queue.popleft()
        print(vertex, end=' ')
        for neighbor in graph[vertex]:
            if not visited[neighbor]:
                queue.append(neighbor)
                visited[neighbor] = True

Check if a Graph is Bipartite

def is_bipartite(graph):
    color = [-1] * len(graph)
    
    def dfs(vertex, c):
        color[vertex] = c
        for neighbor in graph[vertex]:
            if color[neighbor] == -1:
                if not dfs(neighbor, 1 - c): return False
            elif color[neighbor] == color[vertex]:
                return False
        return True
    
    for i in range(len(graph)):
        if color[i] == -1:
            if not dfs(i, 0):
                return False
    return True

Find the Shortest Path in an Unweighted Graph (BFS)

def shortest_path_unweighted(graph, start, target):
    from collections import deque
    queue = deque([(start, 0)])  # (vertex, distance)
    visited = [False] * len(graph)
    visited[start] = True
    
    while queue:
        vertex, dist = queue.popleft()
        if vertex == target:
            return dist
        for neighbor in graph[vertex]:
            if not visited[neighbor]:
                visited[neighbor] = True
                queue.append((neighbor, dist + 1))
    return -1

Dijkstraâ€™s Algorithm

import heapq

def dijkstra(graph, start):
    distances = {vertex: float('infinity') for vertex in range(len(graph))}
    distances[start] = 0
    pq = [(0, start)]  # (distance, vertex)
    
    while pq:
        current_distance, current_vertex = heapq.heappop(pq)
        
        if current_distance > distances[current_vertex]:
            continue
        
        for neighbor, weight in graph[current_vertex]:
            distance = current_distance + weight
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))
    return distances

Bellman-Ford Algorithm

def bellman_ford(graph, vertices, start):
    distances = [float('inf')] * vertices
    distances[start] = 0
    for _ in range(vertices - 1):
        for u, v, w in graph:
            if distances[u] + w < distances[v]:
                distances[v] = distances[u] + w
    for u, v, w in graph:
        if distances[u] + w < distances[v]:
            print("Graph contains negative weight cycle")
            return None
    return distances

Floyd-Warshall Algorithm

def floyd_warshall(graph, vertices):
    dist = [[float('inf')] * vertices for _ in range(vertices)]
    for u in range(vertices):
        dist[u][u] = 0
    for u, v, w in graph:
        dist[u][v] = w
    for k in range(vertices):
        for i in range(vertices):
            for j in range(vertices):
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
    return dist

Find the Number of Connected Components in a Graph

def count_components(graph):
    visited = [False] * len(graph)
    def dfs(vertex):
        visited[vertex] = True
        for neighbor in graph[vertex]:
            if not visited[neighbor]:
                dfs(neighbor)
    
    count = 0
    for i in range(len(graph)):
        if not visited[i]:
            dfs(i)
            count += 1
    return count

Topological Sorting

def topological_sort(graph):
    visited = [False] * len(graph)
    stack = []
    
    def dfs(v):
        visited[v] = True
        for neighbor in graph[v]:
            if not visited[neighbor]:
                dfs(neighbor)
        stack.append(v)

    for i in range(len(graph)):
        if not visited[i]:
            dfs(i)
    
    return stack[::-1]  # Return reversed stack

Detect a Cycle in a Directed Graph

def detect_cycle_directed(graph):
    visited = [False] * len(graph)
    rec_stack = [False] * len(graph)
    
    def dfs(v):
        visited[v] = True
        rec_stack[v] = True
        for neighbor in graph[v]:
            if not visited[neighbor]:
                if dfs(neighbor):
                    return True
            elif rec_stack[neighbor]:
                return True
        rec_stack[v] = False
        return False
    
    for i in range(len(graph)):
        if not visited[i]:
            if dfs(i):
                return True
    return False

Detect a Cycle in an Undirected Graph

def detect_cycle_undirected(graph):
    visited = [False] * len(graph)
    
    def dfs(v, parent):
        visited[v] = True
        for neighbor in graph[v]:
            if not visited[neighbor]:
                if dfs(neighbor, v):
                    return True
            elif parent != neighbor:
                return True
        return False
    
    for i in range(len(graph)):
        if not visited[i]:
            if dfs(i, -1):
                return True
    return False

Primâ€™s Algorithm (Minimum Spanning Tree)

import heapq

def prim(graph):
    mst = []
    visited = [False] * len(graph)
    pq = [(0, 0)]  # (weight, vertex)
    
    while pq:
        weight, u = heapq.heappop(pq)
        if visited[u]:
            continue
        visited[u] = True
        mst.append((weight, u))
        
        for v, w in graph[u]:
            if not visited[v]:
                heapq.heappush(pq, (w, v))
    return mst

Kruskalâ€™s Algorithm (Minimum Spanning Tree)

def kruskal(graph, vertices):
    parent = list(range(vertices))
    rank = [0] * vertices
    
    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])
        return parent[x]

    def union(x, y):
        rootX = find(x)
        rootY = find(y)
        if rootX != rootY:
            if rank[rootX] > rank[rootY]:
                parent[rootY] = rootX
            else:
                parent[rootX] = rootY
                if rank[rootX] == rank[rootY]:
                    rank[rootY] += 1
        return rootX != rootY

    edges = sorted(graph, key=lambda x: x[2])
    mst = []
    
    for u, v, w in edges:
        if union(u, v):
            mst.append((u, v, w))
    return mst

Find All Paths from Source to Destination in a Directed Graph

def all_paths(graph, start, end, path=[]):
    path = path + [start]
    if start == end:
        return [path]
    paths = []
    for neighbor in graph[start]:
        if neighbor not in path:
            new_paths = all_paths(graph, neighbor, end, path)
            for p in new_paths:
                paths.append(p)
    return paths

Word Ladder (Shortest Transformation Sequence from Start to End)

from collections import deque

def word_ladder(start, end, word_list):
    word_set = set(word_list)
    if end not in word_set:
        return 0
    queue = deque([(start, 1)])  # (word, distance)
    
    while queue:
        word, length = queue.popleft()
        if word == end:
            return length
        for i in range(len(word)):
            for c in 'abcdefghijklmnopqrstuvwxyz':
                next_word = word[:i] + c + word[i+1:]
                if next_word in word_set:
                    queue.append((next_word, length + 1))
                    word_set.remove(next_word)
    return 0

Count the Number of Islands in a Matrix (DFS/BFS)

def num_islands(grid):
    def dfs(i, j):
        if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[0]) or grid[i][j] == '0':
            return
        grid[i][j] = '0'  # Mark the cell as visited
        for x, y in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            dfs(i + x, j + y)
    
    island_count = 0
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == '1':
                dfs(i, j)
                island_count += 1
    return island_count

Clone a Graph

class Node:
    def __init__(self, val, neighbors=None):
        self.val = val
        self.neighbors = neighbors or []

def clone_graph(node):
    if not node:
        return None
    
    visited = {}
    def dfs(node):
        if node in visited:
            return visited[node]
        clone = Node(node.val)
        visited[node] = clone
        for neighbor in node.neighbors:
            clone.neighbors.append(dfs(neighbor))
        return clone
    
    return dfs(node)
